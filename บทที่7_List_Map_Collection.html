1. 🖼️ ภาพเล่าเรื่อง: ชั้นวางหนังสือ และตู้ล็อกเกอร์

📚 List = ชั้นวางหนังสือเรียงตามเลขช่อง 🔐 Map = ตู้ล็อกเกอร์ติดป้ายชื่อ ช่อง: [0] [1] [2] ป้าย(key) → ของข้างใน(value) ┌───────┬───────┬───────┐ ┌──────────────────────┐ │ 'ข้อ1' │ 'ข้อ2' │ 'ข้อ3' │ │ 'name' → 'มายด์' │ └───────┴───────┴───────┘ │ 'score' → 42 │ หยิบด้วยเลขช่อง: questions[0] │ 'level' → 'gold' │ └──────────────────────┘ เหมาะกับ: ของหลายชิ้น "ชนิดเดียวกัน" หยิบด้วยป้าย: player['name'] ที่ลำดับมีความหมาย เหมาะกับ: คุณสมบัติหลายด้าน "ของสิ่งเดียว"
⚠️ กับดักเดียวที่ต้องจำจากภาพนี้

ช่องของ List เริ่มนับที่ 0 ไม่ใช่ 1 — หนังสือเล่มแรกอยู่ช่อง [0] เล่มสุดท้ายของชั้นที่มี 10 เล่มอยู่ช่อง [9] สาเหตุอันดับหนึ่งของ error ในบทนี้คือลืมเรื่องนี้

2. 💻 ลงมือจริง: Quiz เพิ่มคำถามได้ไม่จำกัด

import 'dart:io';

void main() {
  // List<String> = ชั้นวางที่รับเฉพาะข้อความ (ระบุชนิดของที่เก็บใน <>)
  final List<String> questions = [
    'Flutter ใช้ภาษาอะไรเขียน?',
    'ไฟล์รายการ package ชื่ออะไร?',
    'คำสั่งดูสถานะ git คืออะไร?',
  ];
  final List<String> answers = ['dart', 'pubspec.yaml', 'git status'];

  int score = 0;

  for (int i = 0; i < questions.length; i++) {
    print('\nข้อ ${i + 1}: ${questions[i]}');       // ผู้เล่นเห็น "ข้อ 1" แต่ index คือ 0
    final userAnswer = (stdin.readLineSync() ?? '').trim().toLowerCase();
    if (userAnswer == answers[i]) {
      score++;
      print('ถูกต้อง! 🎉');
    } else {
      print('เฉลย: ${answers[i]}');
    }
  }
  print('\nสรุป: $score จาก ${questions.length} คะแนน');
}

อยากเพิ่มคำถามที่ 4? เพิ่ม 1 บรรทัดใน List — ลูปจัดการที่เหลือให้หมด นี่คือพลังของ Collection + ลูปที่เรียนบทที่ 5

คำสั่งประจำตัวของ List

final tasks = <String>['ซื้อนม', 'อ่านหนังสือ'];   // อีกท่าประกาศ List
tasks.add('ออกกำลังกาย');    // เพิ่มท้ายชั้น → 3 ชิ้น
tasks.removeAt(0);           // เอาช่อง 0 ออก → เหลือ 2 ชิ้น (ทุกชิ้นขยับซ้าย!)
print(tasks.length);         // 2
print(tasks.contains('อ่านหนังสือ'));  // true
print(tasks.isEmpty);        // false — ใช้เช็คก่อนแสดงผลเสมอ

Map — โปรไฟล์ผู้เล่น

// Map<String, dynamic> = ป้ายเป็นข้อความ ของข้างในเป็นอะไรก็ได้
final Map<String, dynamic> player = {
  'name': 'มายด์',
  'score': 42,
  'isVip': true,
};
print(player['name']);            // มายด์
player['score'] = 50;             // แก้ค่าในล็อกเกอร์
print(player['email']);           // null! — ไม่มีล็อกเกอร์ป้ายนี้ (ไม่ crash แต่ต้องระวัง)
print(player['email'] ?? 'ไม่ระบุ'); // ท่าปลอดภัยจากบทที่ 4 กลับมาอีกแล้ว

.where() และ .map() — เครื่องคัดกรองและเครื่องแปลงร่าง

final scores = [85, 42, 90, 55, 78];

// where = เครื่องคัดกรอง: เก็บเฉพาะชิ้นที่ผ่านเกณฑ์
final passed = scores.where((s) => s >= 60).toList();   // [85, 90, 78]

// map = เครื่องแปลงร่าง: แปลงทุกชิ้นเป็นรูปแบบใหม่
final labels = scores.map((s) => 'ได้ $s คะแนน').toList();
// ['ได้ 85 คะแนน', 'ได้ 42 คะแนน', ...]

อ่านท่า (s) => s >= 60 ว่า: "รับของชิ้นหนึ่งชื่อ s แล้วตอบว่า s ≥ 60 ไหม" — มันคือฟังก์ชันจิ๋วไม่มีชื่อ (ความรู้บทที่แล้ว ในร่างย่อส่วน)

3. 🧠 คอนเซปต์: List คือรากฐานตรงของหน้าจอแอปเกือบทุกหน้า

มองแอปในมือถือคุณตอนนี้: ฟีด Facebook = List ของโพสต์ · แชท Line = List ของข้อความ · YouTube = List ของวิดีโอ — หน้าจอแอปส่วนใหญ่ในโลกคือ "List ที่ถูกวาดเป็นภาพ"

เฟส 2 (ตอนนี้) เฟส 3 (บทที่ 13) List<String> tasks ────────► ListView บนหน้าจอมือถือ วนด้วย for แล้ว print วนด้วย ListView.builder แล้ววาด Widget

สิ่งที่คุณฝึกวันนี้บน Terminal คือกล้ามเนื้อเดียวกับที่จะใช้สร้างหน้าจอจริงเป๊ะ แค่เปลี่ยนจาก print เป็น Widget

4. 🔧 เทคนิคการประยุกต์ใช้

เทคนิคที่ 1 — ตั้งชื่อ List เป็นพหูพจน์เสมอ

tasks, questions, users — เห็นชื่อรู้ทันทีว่าเป็นของหลายชิ้น ส่วนของชิ้นเดียวในลูปใช้เอกพจน์: for (final task in tasks)

เทคนิคที่ 2 — เช็ค isEmpty ก่อนใช้เสมอ

List ว่างคือเรื่องปกติ (ตะกร้ายังไม่มีสินค้า, ยังไม่มีแชท) โค้ดที่ดีเช็คก่อน — ในเฟส 3 นี่จะกลายเป็น "Empty State" หน้าจอสวยๆ ที่แอปดีๆ ทุกตัวมี

เทคนิคที่ 3 — for-in เมื่อไม่ต้องใช้เลขช่อง, for ธรรมดาเมื่อต้องใช้

แค่หยิบของทุกชิ้น → for (final q in questions) สั้นกว่าปลอดภัยกว่า · ต้องรู้ว่าชิ้นที่เท่าไหร่ (แสดง "ข้อ 3") → for (int i = ...) แบบในตัวอย่าง

เทคนิคที่ 4 — ต่อท่อ where + map ได้เหมือนสายพานโรงงาน

scores.where((s) => s >= 60).map((s) => 'ผ่าน: $s').toList() — คัดกรองก่อน แล้วค่อยแปลงร่าง อ่านจากซ้ายไปขวาเหมือนเล่าขั้นตอน นี่คือสไตล์ที่โค้ด Flutter จริงใช้หนาแน่นมาก

5. 🏭 ตัวอย่างการเขียนจริงในงานจริง

🏭 สถานการณ์จริง: "หน้าค้นหาสินค้า"

ทุกครั้งที่คุณพิมพ์ค้นหาในแอปช้อปปิ้ง เบื้องหลังคือ .where() ตรงๆ

// โครงเดียวกับโค้ดหลังบ้านของหน้าค้นหาจริง
final products = ['iPhone 17', 'iPad Pro', 'MacBook Air', 'AirPods Pro'];

List<String> search(String keyword) {
  return products
      .where((p) => p.toLowerCase().contains(keyword.toLowerCase()))
      .toList();
}

// search('pro') → ['iPad Pro', 'AirPods Pro']
// (สังเกต normalize ด้วย toLowerCase ทั้งสองฝั่ง — เทคนิคบทที่ 5 อยู่ทุกที่จริงๆ)

และ .map() คือหัวใจของทุกหน้าจอที่แสดงข้อมูล: เซิร์ฟเวอร์ส่งข้อมูลดิบมา 100 รายการ → .map() แปลงเป็นการ์ดสวยๆ 100 ใบ — ตอนถึงเฟส 4 คุณจะเขียนบรรทัดแบบนี้กับข้อมูลจริงจากอินเทอร์เน็ต

6. 🧹 ตัวอย่าง Clean Code

// ❌ ตัวแปรเบิ้ล — เพิ่มข้อมูลใหม่ต้องแก้โค้ดทุกจุด
final question1 = 'Flutter ใช้ภาษาอะไร?';
final question2 = 'ไฟล์รายการ package ชื่ออะไร?';
final question3 = 'คำสั่งดูสถานะ git คืออะไร?';
print(question1); print(question2); print(question3);
// เพิ่มข้อ 4 = เพิ่มตัวแปร + เพิ่ม print + แก้ตัวนับคะแนน + ... ลืมสักจุดคือบั๊ก
// ✅ List เดียว — โครงสร้างข้อมูลทำงานแทนเรา
final questions = [
  'Flutter ใช้ภาษาอะไร?',
  'ไฟล์รายการ package ชื่ออะไร?',
  'คำสั่งดูสถานะ git คืออะไร?',
  // เพิ่มข้อใหม่: วางบรรทัดเดียวตรงนี้ จบ — ลูป/ตัวนับ/สรุปผล ถูกอัตโนมัติ
];
for (final q in questions) {
  print(q);
}
🧠 สัญญาณเตือน

เมื่อไหร่ที่ตั้งชื่อตัวแปรลงท้ายด้วยเลข (item1, item2, item3) — หยุด แล้วถามตัวเองว่า "นี่ควรเป็น List ไหม?" คำตอบคือใช่เกือบ 100%

7. ⚠️ ข้อผิดพลาดที่เจอบ่อย

① "RangeError (index): Invalid value: Not in inclusive range 0..2: 3"

สาเหตุ: หยิบของเกินชั้น — List มี 3 ชิ้น (ช่อง 0,1,2) แต่ขอช่อง [3] มักเกิดจากลืมว่า index เริ่มที่ 0 หรือใช้ <= ในลูปแทน <
วิธีแก้: ช่องสุดท้ายคือ length - 1 เสมอ · ลูปมาตรฐาน: i < list.length (ไม่ใช่ <=)

② "Concurrent modification during iteration"

สาเหตุ: ลบ/เพิ่มของบนชั้น ขณะที่สายพานกำลังวนหยิบของอยู่ (เช่น removeAt ใน for-in)
วิธีแก้: อย่าแก้ชั้นที่กำลังวน — ใช้ .where() สร้างชั้นใหม่ที่คัดแล้วแทน: tasks = tasks.where((t) => t != 'งานที่จะลบ').toList();

③ ลืม .toList() แล้วผลลัพธ์ใช้ต่อไม่ได้

สาเหตุ: .where()/.map() คืนของชนิด Iterable (สายพานที่ยังไหลอยู่) ไม่ใช่ List
วิธีแก้: ต่อ .toList() ปิดท้ายเมื่ออยากได้ List จริงๆ — จำเป็นนิสัย: where/map แล้วปิดด้วย toList()

8. 🏋️ แบบฝึกหัดท้ายบท (ลงมือทำจริง)

🥉 ระดับ 1 — ชั้นวางของฉัน

สร้าง List<String> รายการสิ่งที่ต้องทำ 5 อย่าง แล้ว (1) พิมพ์ทั้งหมดพร้อมเลขข้อด้วย for (2) .add() เพิ่ม 1 อย่าง (3) .removeAt(0) แล้วสังเกตว่าเลขข้อขยับยังไง (4) พิมพ์เฉพาะรายการที่มีคำว่า "อ่าน" ด้วย .where()

🥈 ระดับ 2 — ตู้ล็อกเกอร์

สร้าง Map<String, dynamic> โปรไฟล์ตัวเอง 5 ฟิลด์ พิมพ์แนะนำตัวจาก Map, ลองอ่าน key ที่ไม่มีแล้วกันด้วย ??, แล้วสร้าง List<Map> เก็บโปรไฟล์เพื่อน 3 คน วนพิมพ์ทุกคนด้วย for-in

9. ✅ เช็กลิสต์และสรุปบทเรียน

📖 สรุปศัพท์ใหม่บทนี้

ศัพท์ความหมายสั้นๆภาพจำ
Listเก็บของหลายชิ้นเรียงลำดับชั้นวางหนังสือเรียงเลขช่อง
Mapเก็บของแบบ ป้าย→ค่าตู้ล็อกเกอร์ติดป้ายชื่อ
indexเลขช่องใน List (เริ่มที่ 0!)เลขช่องบนชั้น
key / valueป้าย / ของข้างใน ของ Mapป้ายล็อกเกอร์ / ของในล็อกเกอร์
for-inวนหยิบของทีละชิ้นโดยไม่สนเลขช่องสายพานอัตโนมัติ
.where()คัดกรองเฉพาะชิ้นที่ผ่านเกณฑ์เครื่องคัดกรอง
.map()แปลงทุกชิ้นเป็นรูปแบบใหม่เครื่องแปลงร่าง
🎓 สรุปบทนี้ใน 3 บรรทัด

List = ชั้นวาง (index เริ่ม 0) · Map = ล็อกเกอร์ (key→value) · where คัดกรอง map แปลงร่าง · เห็นตัวแปรลงท้ายด้วยเลขเมื่อไหร่ = ควรเป็น List
