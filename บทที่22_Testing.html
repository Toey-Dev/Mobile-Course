🥄 Unit Test — ชิมวัตถุดิบทีละอย่าง 📱 Widget Test — ชิมทั้งจานที่จัดเสร็จ "น้ำจิ้มถ้วยนี้เผ็ดถูกสูตรไหม?" "จานนี้หน้าตาถูกต้อง กินได้จริงไหม?" ทดสอบ: ฟังก์ชัน/model/provider ทดสอบ: หน้าจอจริง (แต่ในหน่วยความจำ (Dart ล้วน ไม่มีหน้าจอ) ไม่ต้องเปิดมือถือ!) เช่น: task.toggle() แล้ว isDone เช่น: กดปุ่ม + แล้วมีรายการเพิ่มบนจอไหม กลับด้านจริงไหม ⚡ เร็วมาก (มิลลิวินาที) 🐢 ช้ากว่า (แต่ก็ยังวินาที) ✍️ เขียนง่าย ✍️ เขียนยากกว่านิดหน่อย กฎทองของทีมจริง: เขียน unit test เยอะๆ (ถูกและเร็ว) + widget test เฉพาะจอสำคัญ (แพงกว่า แต่ครอบคลุมกว่า)

จำได้ไหมว่าเราแยก logic ออกจากหน้าจอ (models/, services/, providers/ ไม่รู้จัก Flutter)? — นั่นคือเหตุผลที่บทนี้จะง่ายกว่าที่คิดมาก เพราะ unit test ต้องการโค้ดที่ "ทดสอบได้โดยไม่ต้องเปิดหน้าจอ" ซึ่งคุณเผลอเตรียมไว้ให้ตัวเองแล้วตั้งแต่บทที่แล้ว

2. 💻 ลงมือจริง: เขียนเทสต์ 3 ระดับ

ขั้นที่ 0 — ไขปริศนาที่ค้างมา 21 บท

เปิด my_first_app/test/widget_test.dart — ไฟล์ที่คุณเห็นตั้งแต่วันแรกแต่ยังไม่เคยเข้าใจ:

testWidgets('หน้าแรกแสดงข้อความทักทาย', (WidgetTester tester) async {
  await tester.pumpWidget(const MyApp());              // ① สร้างแอปในหน่วยความจำ
  expect(find.text('สวัสดีชาวโลก! 🌏'), findsOneWidget); // ② ตรวจว่ามีข้อความนี้บนจอ
});

อ่านออกแล้วใช่ไหมครับ: pumpWidget = "ปั๊มแอปขึ้นมาในหน่วยความจำ" (ไม่ต้องเปิด emulator!) · find.text(...) = "หาข้อความนี้บนจอ" · expect(สิ่งที่เจอ, findsOneWidget) = "คาดหวังว่าต้องเจอ 1 อัน" — และจำหมายเหตุบทที่ 1 ได้ไหมที่บอกว่า "ถ้าแก้ข้อความบนจอ ต้องแก้ในเทสต์ด้วย"? นั่นแหละคือเทสต์ทำหน้าที่: ฟ้องทันทีเมื่อหน้าจอเปลี่ยนไปจากที่ตกลงกันไว้

ขั้นที่ 1 — Unit Test ตัวแรก: ทดสอบ model

สร้าง todo_app/test/task_test.dart:

import 'package:flutter_test/flutter_test.dart';
import 'package:todo_app/models/task.dart';

void main() {
  // group = จัดหมวดเทสต์ที่เกี่ยวข้องกัน (เหมือนหมวดในเมนู)
  group('Task', () {
    test('งานใหม่ต้องเริ่มด้วยสถานะยังไม่เสร็จ', () {
      final task = Task(title: 'อ่านหนังสือ');       // 🅰️ Arrange: เตรียมของ
      expect(task.isDone, false);                   // 🅾️ Assert: ตรวจผล
    });

    test('toggle ต้องสลับสถานะเสร็จ/ไม่เสร็จ', () {
      final task = Task(title: 'อ่านหนังสือ');       // 🅰️ Arrange
      task.toggle();                                // 🅱️ Act: ลงมือทำ
      expect(task.isDone, true);                    // 🅾️ Assert
      task.toggle();                                // ทำอีกครั้ง
      expect(task.isDone, false);                   // ต้องกลับมาเหมือนเดิม
    });

    test('fromJson ต้องอ่านข้อมูลเก่าที่ไม่มี isDone ได้', () {
      // เทสต์ "เกราะข้ามเวอร์ชัน" จากบทที่ 18 — พิสูจน์ว่าเกราะทำงานจริง!
      final task = Task.fromJson({'title': 'งานเก่า'});
      expect(task.isDone, false);                   // ต้องได้ค่าตั้งต้น ไม่ใช่ crash
    });
  });
}

ขั้นที่ 2 — โครง Arrange-Act-Assert (AAA): สูตรเขียนเทสต์ที่ใช้ได้ทั้งชีวิต

🅰️ Arrange (เตรียม) — สร้างของที่จะทดสอบ + ตั้งสถานการณ์ 🅱️ Act (ลงมือ) — เรียกสิ่งที่อยากทดสอบ 1 อย่าง 🅾️ Assert (ตรวจ) — เทียบผลลัพธ์กับสิ่งที่คาดหวัง จำง่ายๆ: "เตรียมของ → ลงมือทำ → ตรวจผล" เหมือนการทดลองวิทยาศาสตร์

ขั้นที่ 3 — ทดสอบ Provider (หัวใจของแอป — ตรงนี้คุ้มที่สุด)

// todo_app/test/task_provider_test.dart
group('TaskProvider', () {
  test('addTask ต้องเพิ่มงานเข้า list', () {
    final provider = TaskProvider();                 // 🅰️
    provider.addTask('งานใหม่');                     // 🅱️
    expect(provider.tasks.length, 1);                // 🅾️
    expect(provider.tasks.first.title, 'งานใหม่');
  });

  test('doneCount ต้องนับเฉพาะงานที่เสร็จแล้ว', () {
    final provider = TaskProvider();                 // 🅰️
    provider.addTask('งาน A');
    provider.addTask('งาน B');
    provider.toggleTask(provider.tasks.first);       // 🅱️ ติ๊กแค่ตัวแรก
    expect(provider.doneCount, 1);                   // 🅾️
  });

  test('addTask ต้องปฏิเสธชื่อซ้ำ', () {
    final provider = TaskProvider();                 // 🅰️
    provider.addTask('งานซ้ำ');
    final error = provider.addTask('งานซ้ำ');        // 🅱️ เพิ่มชื่อเดิมอีกครั้ง
    expect(error, isNotNull);                        // 🅾️ ต้องได้ข้อความ error
    expect(provider.tasks.length, 1);                //    และต้องไม่เพิ่มเข้าไปจริง
  });
});
🎯 จุดที่การจัดโครงสร้างจ่ายเงินปันผลก้อนใหญ่

TaskProvider เป็น Dart ล้วน ทดสอบได้โดยไม่ต้องเปิดหน้าจอสักบรรทัด — ถ้าตรรกะพวกนี้ยังฝังอยู่ในปุ่มบนหน้าจอ (แบบ ❌ ของบทที่ 17) การทดสอบจะยากกว่านี้หลายเท่า

ขั้นที่ 4 — Widget Test: ทดสอบหน้าจอจริง

// todo_app/test/home_screen_test.dart
testWidgets('กดปุ่ม + แล้วต้องมีรายการเพิ่มบนจอ', (tester) async {
  // 🅰️ Arrange: ปั๊มแอปขึ้นมา (ต้องครอบ Provider เหมือนใน main.dart!)
  await tester.pumpWidget(
    ChangeNotifierProvider(
      create: (_) => TaskProvider(),
      child: const TodoApp(),
    ),
  );

  expect(find.text('ยังไม่มีงาน 🎉'), findsOneWidget);  // เริ่มต้นต้องเป็น Empty State

  // 🅱️ Act: กดปุ่มลอย
  await tester.tap(find.byIcon(Icons.add));
  await tester.pump();       // ⭐ สำคัญ! สั่งให้ Flutter วาดจอใหม่หลังเกิดเหตุ

  // 🅾️ Assert: Empty State ต้องหายไป
  expect(find.text('ยังไม่มีงาน 🎉'), findsNothing);
});
⭐ pump() คือหัวใจที่มือใหม่ลืมบ่อยที่สุด

ในเทสต์ เวลาไม่เดินเอง! หลังทุก action ที่ทำให้จอเปลี่ยน (tap, enterText) ต้องสั่ง pump() เพื่อบอกว่า "วาดเฟรมใหม่ทีนะ" ไม่งั้นคุณจะตรวจจอเก่าที่ยังไม่อัปเดต

3. 🔧 เทคนิคการประยุกต์ใช้

เทคนิคที่ 1 — ชื่อเทสต์คือประโยคบอกเล่าที่คนอ่านเข้าใจ

test('toggle ต้องสลับสถานะเสร็จ/ไม่เสร็จ') ✓ ไม่ใช่ test('test1') ✗ — เพราะเวลาเทสต์แดง สิ่งแรกที่คุณเห็นคือชื่อมัน ชื่อที่ดีบอกทันทีว่า "พฤติกรรมอะไรพัง" โดยไม่ต้องเปิดโค้ด (หลักเดียวกับชื่อฟังก์ชันบทที่ 6)

เทคนิคที่ 2 — เขียนเทสต์ที่ "เคยพัง" ก่อนเสมอ

ไม่ต้องเทสต์ทุกบรรทัดในโลก — เริ่มจาก: (1) ตรรกะที่คำนวณอะไรสักอย่าง (doneCount, ราคารวม) (2) กฎธุรกิจ (ห้ามชื่อซ้ำ) (3) บั๊กที่เคยเจอจริง — เจอบั๊กแล้วแก้เสร็จ ให้เขียนเทสต์ปิดท้าย เพื่อรับประกันว่า "บั๊กตัวนี้จะไม่กลับมาอีกตลอดกาล" (ทีมจริงเรียกว่า regression test — คู่แฝดของ BUGS.md จากบทที่ 14)

เทคนิคที่ 3 — find มีหลายวิธี เลือกให้เหมาะ

find.text('บันทึก') (หาจากข้อความ), find.byIcon(Icons.add) (หาจากไอคอน), find.byType(TaskTile) (หาจากชนิด Widget) — ตัวหลังทนต่อการเปลี่ยนข้อความมากที่สุด เหมาะกับเทสต์ที่ไม่อยากพังทุกครั้งที่แก้คำ

เทคนิคที่ 4 — เทสต์ที่ดีคือเอกสารที่ไม่มีวันล้าสมัย

README อาจเขียนไว้ผิดโดยไม่มีใครรู้ แต่เทสต์ที่บอกว่า "addTask ต้องปฏิเสธชื่อซ้ำ" ถูกเสมอ เพราะถ้าโค้ดเปลี่ยนพฤติกรรม เทสต์จะแดงทันที — dev ใหม่ในทีมมักอ่านเทสต์เพื่อทำความเข้าใจว่าระบบทำงานยังไง

4. 🏭 ตัวอย่างการเขียนจริงในงานจริง

🏭 สถานการณ์จริง: "ยามเฝ้าประตูที่ชื่อ CI"

ในทีมจริง เทสต์ไม่ได้รันด้วยมือ — มันรันอัตโนมัติทุกครั้งที่มีคนส่งโค้ด:

dev ส่ง Pull Request │ ▼ 🤖 CI (เครื่องอัตโนมัติบนคลาวด์) ทำงานทันที: ① flutter analyze (ตรวจ lint — กฎที่เราตั้งไว้ตั้งแต่บทที่ 2!) ② flutter test (ชิมทุกเมนู) │ ┌─────┴─────┐ ✅ เขียว ❌ แดง │ │ │ └─► ปุ่ม "Merge" ถูกล็อก — เข้า main ไม่ได้จนกว่าจะแก้ │ (ยามไม่ปล่อยผ่าน ไม่ว่าคุณจะเป็นใครในทีม) └─► เพื่อนรีวิว → merge เข้า main ได้ ผลลัพธ์: เส้น main "ใช้งานได้เสมอ" จริงตามที่บทที่ 2 บอกไว้ — เพราะมียามเฝ้า ไม่ใช่เพราะทุกคนระวังตัว

และนี่คือเหตุผลที่ dev มืออาชีพชอบเขียนเทสต์ ทั้งที่มันเป็นงานเพิ่ม: มันคือสิ่งที่ทำให้คุณ กล้าแก้โค้ดเก่าที่ไม่ได้เขียนเอง — รื้อได้เต็มที่ ถ้าเทสต์ยังเขียว แปลว่าไม่พัง (เหมือนจุดเซฟในเกมจากบทที่ 2 แต่เป็นจุดเซฟของ "พฤติกรรมแอป")

5. 🧹 Clean Code: เทสต์ที่ทดสอบ "พฤติกรรม" ไม่ใช่ "วิธีทำ"

  expect(provider.tasks.map((t) => t.title), contains('งาน'));

ถามตัวเองว่า "ถ้าฉัน refactor ข้างในโดยที่แอปทำงานเหมือนเดิมทุกอย่าง เทสต์นี้ควรแดงไหม?" — ถ้าตอบว่า "ไม่ควร" แต่มันแดง = เทสต์ผูกลึกเกินไป · เทสต์ที่ดีคือสัญญาว่าแอปจะทำอะไรได้ ไม่ใช่บันทึกว่าโค้ดเขียนยังไง

6. ⚠️ ข้อผิดพลาดที่เจอบ่อย

① "Could not find the correct Provider above this Widget" ในเทสต์

สาเหตุ: pumpWidget แอปเปล่าๆ โดยลืมครอบ ChangeNotifierProvider — ในเทสต์เราสร้างต้นไม้ Widget เองทั้งหมด สิ่งที่ main.dart ครอบไว้ ไม่ได้ตามมาให้
วิธีแก้: จำลองสภาพแวดล้อมให้เหมือน main.dart: ครอบ provider (และ MaterialApp ถ้าเทสต์หน้าจอย่อยที่ต้องใช้ Navigator/Theme) — เพื่อนเก่าจากบทที่ 17

② เทสต์แดงทั้งที่โค้ดถูก — เพราะลืม pump()

สาเหตุ: หลัง tester.tap() แล้วตรวจผลทันที — แต่ Flutter ยังไม่ได้วาดเฟรมใหม่ (ในเทสต์ เวลาไม่เดินเอง! เราต้องสั่งเดินเอง)
วิธีแก้: ทุก action ที่ทำให้จอเปลี่ยน ต้องตามด้วย await tester.pump(); ถ้ามีแอนิเมชัน/รอ Future ใช้ await tester.pumpAndSettle(); (รอจนนิ่ง)

③ เทสต์ผ่านบ้างไม่ผ่านบ้าง (flaky test) หรือทดสอบของจริงบนเน็ต

สาเหตุ: เทสต์ยิง API จริง — เน็ตช้า/เซิร์ฟเวอร์ล่ม = เทสต์แดงทั้งที่โค้ดเราถูก (และช้ามาก + ถ้าเป็น API ที่คิดเงิน คือจ่ายเงินทุกครั้งที่รันเทสต์!)
วิธีแก้: เทสต์ต้อง "ปิดเกาะ" — ไม่พึ่งพาโลกภายนอก ให้ป้อนข้อมูลปลอมเข้าไปแทน (เทคนิคนี้เรียกว่า mocking — เนื้อหาระดับถัดไป ตอนนี้แค่รู้ว่า "เทสต์ที่ดีต้องรันได้แม้ปิดเน็ต" ก็พอ และเป็นเหตุผลที่เราเทสต์ provider/model ก่อน)

7. 🏋️ แบบฝึกหัดท้ายบท (ลงมือทำจริง)

🥉 ระดับ 1 — พนักงานชิมคนแรก

เขียน test/task_test.dart ตามขั้นที่ 1 ให้ครบ 3 เทสต์ แล้วรัน flutter test จนเขียว — จากนั้นทำให้มันแดงโดยตั้งใจ: ไปแก้ toggle() ใน model ให้ผิด แล้วรันใหม่ อ่านรายงานที่บอกว่าเทสต์ไหนพัง คาดหวังอะไร ได้อะไร — แล้วแก้กลับ (การเห็นเทสต์แดงคือส่วนสำคัญของการเรียนรู้!)

🥈 ระดับ 2 — ชิมหัวใจของแอป

เขียน test/task_provider_test.dart ตามขั้นที่ 3 ครบ 3 เทสต์ + เพิ่มอีก 2 เทสต์ของตัวเอง (เช่น removeTask ต้องลบออกจริง, งานเกิน 50 ต้องปฏิเสธ) — commit ด้วย test: add unit tests for Task and TaskProvider

🥇 ระดับ 3 — ชิมทั้งจาน + ยามเฝ้าประตู

(1) เขียน widget test อย่างน้อย 2 ตัวสำหรับ todo_app: กดปุ่มเพิ่มแล้วจอเปลี่ยน, ติ๊ก checkbox แล้วข้อความถูกขีดฆ่า (ใช้ find.byType) (2) เขียน unit test ให้ Weather.fromJson ใน weather_app โดยป้อน JSON ปลอมที่ก็อปมาจาก browser (บทที่ 19 ขั้นที่ 0!) — พิสูจน์ว่า (as num).toDouble() รับได้ทั้ง 32 และ 32.5 (3) เพิ่มหัวข้อ "การทดสอบ" ใน README ทั้ง 3 repo — commit ด้วย test: widget tests and JSON parsing tests

8. ✅ เช็กลิสต์และสรุปบทเรียน

  • อ่าน widget_test.dart ที่มีมาแต่แรกแล้วอธิบายได้ทุกบรรทัด
  • บอกความต่าง unit test กับ widget test และเลือกใช้ถูกสถานการณ์
  • เขียนเทสต์ตามโครง Arrange-Act-Assert ได้เอง
  • รู้ว่าต้อง pump() หลังทุก action ในเทสต์ และรู้ว่าทำไม
  • อธิบายได้ว่าเทสต์ควรผูกกับ "พฤติกรรม" ไม่ใช่ "วิธีทำข้างใน"
  • flutter test ผ่านเขียวทั้ง 3 repo และเคยเห็นเทสต์แดงจากการตั้งใจทำพัง
